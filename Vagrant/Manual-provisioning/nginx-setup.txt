#installing the nginx server

sudo apt update && apt upgrade 

sudo apt install nginx -y

vim /etc/sites-available/vproapp

upstream vproapp {
    server app01:8080;
}
server {
    listen 80;
    location / {
        proxy_pass http://vproapp
    }
}

rm -rf /etc/sites-enabled/default

ln -s /etc/sites-available/vproapp /etc/sites-enabled/vproapp
